Space Adventure

Dive deeper into the code, mechanics, and creative process behind my Space Adventure project.

Project Overview

Space Adventure is a modern reimagining of the classic Space Invaders game. Built in JavaScript with p5.js, it combines classic arcade mechanics with upgrades like shields, UFOs, and dynamic difficulty.

Gameplay Mechanics

The core player interactions: moving left/right and shooting bullets, plus invader movement and attacks.


function keyboardControls() {
  if (keyIsDown(LEFT_ARROW)) player.move(true);
  if (keyIsDown(RIGHT_ARROW)) player.move(false);
  if (keyIsDown(32)) player.shoot();
}
function createBullet(x, y, img) {
  let bulletPrefab = new Sprite(x, y, img.width*scale, img.height*scale, img);
  bulletPrefab.vel = createVector(0, -scale*15);
  bulletPrefab.update = function() { this.pos.add(this.vel); };
  return bulletPrefab;
}

Collision Detection

This system checks collisions between bullets, players, invaders, UFOs, and shields, applying damage and updating scores.


function checkAllCollisions() {
  invaderBullets.forEach(bullet => {
    if (!player.invincible && player.intersects(bullet)) {
      playerHitSound.play();
      hit();
      bullet.deadMarked = true;
    }
  });
  bullets.forEach(bullet => {
    invaders.forEach(invader => {
      if (bullet.intersects(invader)) {
        invader.health -= bullet.damage;
        bullet.deadMarked = true;
      }
    });
  });
}

Wave & Difficulty System

Dynamic waves with increasing difficulty, health, and invader shoot speed.


function makeWave(difficulty) {
  let config = [];
  switch(difficulty) {
    case 2: config = [{canShoot:false,extraHealth:0},{canShoot:true,extraHealth:0}]; break;
    case 3: config = [{canShoot:false,extraHealth:0},{canShoot:true,extraHealth:0}]; break;
    default: config = [{canShoot:false,extraHealth:0},{canShoot:true,extraHealth:0}];
  }
  return config;
}
function dificultyHigher() {
  invaderHealth++;
  MOVE_INTERVAL = max(MOVE_INTERVAL-1, 1);
  dificultyLevel++;
  enemyShootFaster();
}
